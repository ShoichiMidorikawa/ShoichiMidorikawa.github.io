サイン関数のグラフ
sin.pltファイル

# サイン関数の描画
reset
set terminal png
set output "sin.png"
set xtics ( "-3p" -3*pi, "-2p" -2*pi, "-p" -pi, \
"0" 0.0, "p" pi, "2p" 2*pi, "2p" 2*pi, "3p" 3*pi) font "Symbol"
set xzeroaxis
set yzeroaxis
plot sin(x)

lissajous.pltファイル

# リサージュ図形(Lissajous figure)の描画
reset
set terminal png
set output "lissajous.png"
set size square
set parametric
a=1
b=2
plot sin(a*t), sin(b*t)
